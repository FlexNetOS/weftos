Expand description

A tool that can be invoked by the agent pipeline.

Implementations provide a name, description, JSON Schema for parameters, and an async execute method. Tools are registered in a ToolRegistry and dispatched by the agent loop when the LLM emits a tool-use request.

§Implementing a tool

ⓘ
use async_trait::async_trait;
use clawft_core::tools::registry::{Tool, ToolError};

struct EchoTool;

#[async_trait]
impl Tool for EchoTool {
    fn name(&self) -> &str { "echo" }
    fn description(&self) -> &str { "Echo back the input" }
    fn parameters(&self) -> serde_json::Value {
        serde_json::json!({
            "type": "object",
            "properties": {
                "text": { "type": "string", "description": "Text to echo" }
            },
            "required": ["text"]
        })
    }
    async fn execute(&self, args: serde_json::Value) -> Result<serde_json::Value, ToolError> {
        let text = args.get("text")
            .and_then(|v| v.as_str())
            .ok_or_else(|| ToolError::InvalidArgs("missing 'text'".into()))?;
        Ok(serde_json::json!({ "output": text }))
    }
}
